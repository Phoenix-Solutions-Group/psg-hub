DEP_MAP=(
  "studio:@psg/studio"

  "preview:@psg/preview"
  "shops:@psg/shops"

BSM_PREFIXES=("bsm-" "")

tmp=$(mktemp)

for entry in "${DEP_MAP[@]}"; do
  old_name="${entry%:*}"
  new_name="${entry##*:}"
  for prefix in "${BSM_PREFIXES[@]}"; do
    candidate="${prefix}${old_name}"
    # If candidate key appears in dependencies/devDependencies/peerDependencies, swap
    for section in dependencies devDependencies peerDependencies; do
      has=$(jq --arg s "$section" --arg k "$candidate" '.[$s][$k] // empty' "$tmp")

           "$tmp" > "${tmp}.next" && mv "${tmp}.next" "$tmp"
        echo "REWROTE: $candidate -> $new_name (workspace:*) in $section"
      fi
    done
